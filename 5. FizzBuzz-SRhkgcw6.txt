#include <conio.h>
#include <iostream>
int main()
{
	for (int i = 1; i <= 100; ++i)
	{
		std::cout << i << '\r'; // return to beginning of line for overwriting with "Fizz" or "Buzz"
		if (i % 3 == 0) std::cout << "Fizz";
		if (i % 5 == 0) std::cout << "Buzz";
		std::cout << '\n';
	}
	_getch();
	return 0;
}
