    private void Dodaj_ButtonActionPerformed(java.awt.event.ActionEvent evt) {                                             
       String temp = new String();
       temp = Tekstowe_pole.getText();
        if(Na_poczatku_check.isSelected())
        {
            
            String sel_temp = temp;
            temp=(String)Combo_lista.getSelectedItem();
            temp+=sel_temp;
          
            Tekstowe_pole.setText(temp);
        }
        else
        {
            temp += (String)Combo_lista.getSelectedItem();
            Tekstowe_pole.setText(temp);
        }
        
    }                                            

    private void Dopisz_buttonActionPerformed(java.awt.event.ActionEvent evt) {                                              
        String tekst = Tekstowe_pole.getText();
        
        if(!tekst.equals("")){
            vect.add(tekst);
            Lista_rozwijana.setListData(vect);
            Tekstowe_pole.setText("");
        }
        
    }
